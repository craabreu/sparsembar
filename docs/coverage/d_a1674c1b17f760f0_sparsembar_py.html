20@jax.jit 

21def sparse_mbar_negative_log_likelihood( 

22 free_energies: jnp.ndarray, 

23 potentials: t.Sequence[jnp.ndarray], 

24 sample_sizes: t.Sequence[jnp.ndarray], 

25 state_indices: t.Sequence[jnp.ndarray], 

26) -> float: 

27 """The negative of the log-likelihood function that is maximized by the 

28 Sparse MBAR estimator.""" 

29 free_energies = jnp.insert(free_energies, 0, 0.0) 

30 return sum( 

31 map( 

32 mbar_negative_log_likelihood, 

33 [jnp.take(free_energies, indices) for indices in state_indices], 

34 potentials, 

35 sample_sizes, 

36 it.repeat(False), 

37 ) 

38 )

45class SparseMBAR: # pylint: disable=too-few-public-methods 

46 """ 

47 A class for performing MBAR estimation on groups of states with sparse 

48 connections. 

49 

50 Parameters 

51 ---------- 

52 state_groups 

53 A sequence of state groups. 

54 method 

55 The minimization method to use for free energy calculation. The options are the 

56 same as for :func:`scipy.optimize.minimize`. 

57 tolerance 

58 The tolerance for termination of the minimization. Each method sets some 

59 relevant solver-specific tolerance(s) equal to this value. 

60 allow_unconverged 

61 Whether to allow unconverged minimization results due to lack of numerical 

62 precision. 

63 **kwargs 

64 Additional keyword arguments that will be passed to the 

65 :func:`scipy.optimize.minimize` function, except for ``method``, ``tol``, 

66 ``jac`` and ``hess``. 

67 

68 Examples 

69 -------- 

70 >>> import sparsembar as smbar 

71 >>> state_id_lists = [0, 1, 2], [2, 3, 4] 

72 >>> models = [ 

73 ... smbar.MultiGaussian(state_ids, 1) 

74 ... for state_ids in state_id_lists 

75 ... ] 

76 >>> potentials = [ 

77 ... model.compute_reduced_potentials(model.draw_samples(100)) 

78 ... for model in models 

79 ... ] 

80 >>> estimator = smbar.SparseMBAR( 

81 ... smbar.StateGroup(ids, matrix) 

82 ... for ids, matrix in zip(state_id_lists, potentials) 

83 ... ) 

84 >>> states = estimator.all_states 

85 >>> states 

86 (0, 1, 2, 3, 4) 

87 >>> [estimator.groups_with_state(state) for state in states] 

88 [(0,), (0,), (0, 1), (1,), (1,)] 

89 """

91 def __init__( 

92 self, 

93 state_groups: t.Iterable[StateGroup], 

94 *, 

95 method: str = "BFGS", 

96 tolerance: float = 1e-12, 

97 allow_unconverged: bool = True, 

98 **kwargs, 

99 ) -> None: 

100 self._groups = list(state_groups) 

101 self._num_groups = len(self._groups) 

102 

103 self._groups_with_state = {} 

104 for group_index, group in enumerate(self._groups): 

105 for state_index, state in enumerate(group.states): 

106 if state not in self._groups_with_state: 

107 self._groups_with_state[state] = [] 

108 self._groups_with_state[state].append((group_index, state_index)) 

109 

110 self._all_states = tuple(self._groups_with_state) 

111 self._num_states = len(self._all_states) 

112 

113 self._state_indices = [ 

114 jnp.array([self._all_states.index(state) for state in group.states]) 

115 for group in self._groups 

116 ] 

117 

118 self._rough_free_energies = self._compute_rough_free_energies() 

119 self._free_energies = self._compute_free_energies( 

120 method, tolerance, allow_unconverged, **kwargs 

121 )

123 def _concatenate(self, func: t.Callable[[StateGroup], jnp.ndarray]) -> jnp.ndarray: 

124 zeros = jnp.zeros(self._num_states) 

125 return jnp.vstack( 

126 [ 

127 jnp.put(zeros, state_indices, func(group), inplace=False) 

128 for group, state_indices in zip(self._groups, self._state_indices) 

129 ] 

130 )

132 def _compute_rough_free_energies(self) -> jnp.ndarray: 

133 sample_sizes = self._concatenate(lambda group: group.sample_sizes) 

134 free_energies = self._concatenate(lambda group: group.get_free_energies()) 

135 w_fractions = sample_sizes / sample_sizes.sum(axis=1, keepdims=True) 

136 z_fractions = sample_sizes / sample_sizes.sum(axis=0, keepdims=True) 

137 unshifted_means = (z_fractions * free_energies).sum(axis=0) 

138 alpha_matrix = w_fractions.dot(z_fractions.T) - jnp.identity(self._num_groups) 

139 b_vector = (w_fractions * (free_energies - unshifted_means)).sum(axis=1) 

140 shifts = jnp.linalg.solve( 

141 alpha_matrix.at[0, :].set(z_fractions.at[:, 0].get()), 

142 b_vector.at[0].set(-unshifted_means.at[0].get()), 

143 ) 

144 return unshifted_means + z_fractions.T.dot(shifts) 

145 

146 def _compute_free_energies( 

147 self, method: str, tolerance: float, allow_unconverged: bool, **kwargs 

148 ) -> None: 

149 free_energies = argmin( 

150 sparse_mbar_negative_log_likelihood, 

151 jnp.delete(self._rough_free_energies, 0), 

152 ( 

153 [group.potentials for group in self._groups], 

154 [group.sample_sizes for group in self._groups], 

155 self._state_indices, 

156 ), 

157 method, 

158 tolerance, 

159 allow_unconverged, 

160 sparse_mbar_gradient, 

161 sparse_mbar_hessian, 

162 **kwargs, 

163 ) 

164 return jnp.insert(free_energies, 0, 0.0)

201 def get_free_energies( 

202 self, return_dict: bool = False 

203 ) -> t.Union[t.Dict[t.Hashable, float], t.Tuple[jnp.ndarray, ...]]: 

204 """ 

205 Return the free energies of all states in the groups. 

206 

207 Parameters 

208 ---------- 

209 return_dict 

210 Whether to return a dictionary of free energies with the states as keys. 

211 

212 Returns 

213 ------- 

214 t.Union[t.Dict[t.Hashable, float], jnp.ndarray] 

215 The reduced free energies of the states. If ``return_dict`` is ``True``, 

216 the return value is a dictionary of free energies with the states as keys. 

217 Otherwise, it is an array of free energies. 

218 

219 Examples 

220 -------- 

221 >>> import sparsembar as smbar 

222 >>> mean_lists = [0, 1, 2], [2, 2, 1, 0] 

223 >>> sigma_lists = [1, 2, 3], [3, 2, 1, 1] 

224 

225 Using tuples of means and standard deviations as state IDs: 

226 

227 >>> state_id_lists = [ 

228 ... [(mean, sigma) for mean, sigma in zip(means, sigmas)] 

229 ... for means, sigmas in zip(mean_lists, sigma_lists) 

230 ... ] 

231 >>> state_id_lists 

232 [[(0, 1), (1, 2), (2, 3)], [(2, 3), (2, 2), (1, 1), (0, 1)]] 

233 

234 Sampling from multi-dimensional Gaussian distributions: 

235 

236 >>> models = [ 

237 ... smbar.MultiGaussian(means, sigmas, seed=123) 

238 ... for means, sigmas in zip(mean_lists, sigma_lists) 

239 ... ] 

240 >>> potentials = [ 

241 ... model.compute_reduced_potentials(model.draw_samples(100)) 

242 ... for model in models 

243 ... ] 

244 

245 Estimating the free energies of the states: 

246 

247 >>> estimator = smbar.SparseMBAR( 

248 ... smbar.StateGroup(ids, matrix) 

249 ... for ids, matrix in zip(state_id_lists, potentials) 

250 ... ) 

251 >>> estimator.get_free_energies() 

252 (Array([...], ...dtype=float64), Array([...], ...dtype=float64)) 

253 >>> estimator.get_free_energies(return_dict=True) 

254 {(0, 1): 0.0, (1, 2): ..., (2, 3): ..., (2, 2): ..., (1, 1): ...} 

255 """ 

256 if return_dict: 

257 return dict(zip(self._all_states, self._free_energies.tolist())) 

258 return tuple( 

259 jnp.take(self._free_energies, indices) for indices in self._state_indices 

260 )
